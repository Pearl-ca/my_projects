from scipy import*   
def Eur_price(S0,E,T,N,r,sigma,cp):
    ## cp=1 - call option; cp=-1 - put option
    dt=T/N
    U=exp(sigma*sqrt(dt))
    D=exp(-sigma*sqrt(dt))
    R=(exp(r*dt))
    q=(R-D)/(U-D)
    V_old=zeros(N+1)
    V_new=zeros(N+1)
    for m in range(0,N+1):
        V_old[m]=max(cp*(S0*U**m*D**(N-m)-E),0)    
    for j in range(N-1,-1,-1):
        for m in range(0,j+1):
            V_new[m]=(q*V_old[m+1]+(1-q)*V_old[m])/R
        for m in range(0,j+1):
            V_old[m]=V_new[m]
    return(V_new[0])

from scipy import*   
def Amer_price(S0,E,T,N,r,sigma,cp):
    dt=T/N
    U=exp(sigma*sqrt(dt))
    D=exp(-sigma*sqrt(dt))
    R=(exp(r*dt))
    q=(R-D)/(U-D)
    V_old=zeros(N+1)
    V_new=zeros(N+1)
    for m in range(0,N+1):
        V_old[m]=max(cp*(S0*U**m*D**(N-m)-E),0)    
    for j in range(N-1,-1,-1):
        for m in range(0,j+1):
            V_new[m]=(q*V_old[m+1]+(1-q)*V_old[m])/R
            V_new[m]=max(V_new[m],max(cp*(S0*U**m*D**(j-m)-E),0))
        for m in range(0,j+1):
            V_old[m]=V_new[m]
    return(V_new[0])

##Examples:
print('The American call option price is %8.4f.' % Amer_price(S0=100,E=110,T=1,N=1000,r=0.05,sigma=0.3,cp=1))
print('The American put option price is %8.4f.' % Amer_price(S0=100,E=110,T=1,N=1000,r=0.05,sigma=0.3,cp=-1))
